
#!/bin/bash

echo "ðŸ“¦ Setting up Shadow Framework..."

BASE_DIR="$HOME/shadow_framework"
TOOLS_DIR="$BASE_DIR/tools"
mkdir -p "$TOOLS_DIR"

function check_and_install() {
    if ! command -v $1 &> /dev/null; then
        echo "ðŸ”§ Installing $1..."
        eval "$2"
    else
        echo "âœ… $1 already installed."
    fi
}

check_and_install "whois" "sudo apt install whois -y"
check_and_install "nslookup" "sudo apt install dnsutils -y"
check_and_install "nmap" "sudo apt install nmap -y"
check_and_install "whatweb" "sudo apt install whatweb -y"
check_and_install "sqlmap" "sudo apt install sqlmap -y"
check_and_install "macchanger" "sudo apt install macchanger -y"
check_and_install "tor" "sudo apt install tor -y"
check_and_install "torsocks" "sudo apt install torsocks -y"
check_and_install "amass" "sudo snap install amass"

# Clone tools
if [ ! -d "$TOOLS_DIR/Sublist3r" ]; then
    git clone https://github.com/aboul3la/Sublist3r.git "$TOOLS_DIR/Sublist3r"
fi

if [ ! -d "$TOOLS_DIR/XSStrike" ]; then
    git clone https://github.com/s0md3v/XSStrike.git "$TOOLS_DIR/XSStrike"
    pip install -r "$TOOLS_DIR/XSStrike/requirements.txt"
fi

echo "âœ… Shadow Framework setup complete!"
